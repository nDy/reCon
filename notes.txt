detect(Image i){
	compute Hn by edge extraction
	compute Hn,l from histograms
	for all(l)
		compute feature vectors from histograms
		compute segment S
		for all (S)
			compute probability
			if (P(Si)>= threshold)
				QR detection algorithm
			endif
		end for
	end for
}
